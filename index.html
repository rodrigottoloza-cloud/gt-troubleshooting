const express = require("express"); const cors = require("cors"); const multer = require("multer"); const sqlite3 = require("sqlite3").verbose(); const path = require("path"); const fs = require("fs"); const app = express(); const PORT = 3000; // Rutas base const dataDir = path.join(__dirname, "data"); const uploadsDir = path.join(__dirname, "uploads"); const dbPath = path.join(dataDir, "gt.sqlite"); // Asegurar carpetas if (!fs.existsSync(dataDir)) fs.mkdirSync(dataDir, { recursive: true }); if (!fs.existsSync(uploadsDir)) fs.mkdirSync(uploadsDir, { recursive: true }); // Middlewares app.use(cors()); app.use(express.json()); app.use("/uploads", express.static(uploadsDir)); // Base de datos const db = new sqlite3.Database(dbPath, (err) => { if (err) { console.error("Error abriendo SQLite:", err.message); } else { console.log("SQLite conectado en:", dbPath); } }); // Crear tablas db.serialize(() => { db.run(` CREATE TABLE IF NOT EXISTS users ( id INTEGER PRIMARY KEY AUTOINCREMENT, email TEXT UNIQUE NOT NULL, password TEXT NOT NULL, name TEXT NOT NULL ) `); db.run(` CREATE TABLE IF NOT EXISTS cases ( id INTEGER PRIMARY KEY AUTOINCREMENT, aircraft_model TEXT, serial_number TEXT, ata TEXT, fault_type TEXT, reported_fault TEXT, result_summary TEXT, active_technical_data_name TEXT, created_at TEXT DEFAULT CURRENT_TIMESTAMP ) `); db.run(` CREATE TABLE IF NOT EXISTS uploads ( id INTEGER PRIMARY KEY AUTOINCREMENT, label TEXT, original_name TEXT, stored_name TEXT, mime_type TEXT, size INTEGER, created_at TEXT DEFAULT CURRENT_TIMESTAMP ) `); // Usuario inicial db.run( ` INSERT OR IGNORE INTO users (email, password, name) VALUES (?, ?, ?) `, ["rodrigo@gt-troubleshooting.com", "1234", "Rodrigo Tapia"] ); }); // Multer const storage = multer.diskStorage({ destination: function (req, file, cb) { cb(null, uploadsDir); }, filename: function (req, file, cb) { const uniqueName = `${Date.now()}-${file.originalname.replace(/\s+/g, "_")}`; cb(null, uniqueName); } }); const upload = multer({ storage }); // Health check app.get("/", (req, res) => { res.json({ ok: true, message: "G&T backend funcionando" }); }); // Login simple app.post("/login", (req, res) => { const { email, password } = req.body; if (!email || !password) { return res.status(400).json({ ok: false, message: "Email y contraseña son obligatorios" }); } db.get( `SELECT id, email, name FROM users WHERE email = ? AND password = ?`, [email, password], (err, row) => { if (err) { return res.status(500).json({ ok: false, message: "Error en base de datos" }); } if (!row) { return res.status(401).json({ ok: false, message: "Credenciales inválidas" }); } res.json({ ok: true, user: row }); } ); }); // Guardar caso app.post("/cases", (req, res) => { const { aircraft_model, serial_number, ata, fault_type, reported_fault, result_summary, active_technical_data_name } = req.body; db.run( ` INSERT INTO cases ( aircraft_model, serial_number, ata, fault_type, reported_fault, result_summary, active_technical_data_name ) VALUES (?, ?, ?, ?, ?, ?, ?) `, [ aircraft_model || "", serial_number || "", ata || "", fault_type || "", reported_fault || "", result_summary || "", active_technical_data_name || "" ], function (err) { if (err) { return res.status(500).json({ ok: false, message: "No se pudo guardar el caso" }); } res.json({ ok: true, id: this.lastID, message: "Caso guardado correctamente" }); } ); }); // Listar casos app.get("/cases", (req, res) => { db.all( `SELECT * FROM cases ORDER BY id DESC`, [], (err, rows) => { if (err) { return res.status(500).json({ ok: false, message: "No se pudieron obtener los casos" }); } res.json({ ok: true, cases: rows }); } ); }); // Eliminar caso app.delete("/cases/:id", (req, res) => { const { id } = req.params; db.run(`DELETE FROM cases WHERE id = ?`, [id], function (err) { if (err) { return res.status(500).json({ ok: false, message: "No se pudo eliminar el caso" }); } res.json({ ok: true, deleted: this.changes > 0 }); }); }); // Subir archivo app.post("/upload", upload.single("file"), (req, res) => { const label = req.body.label || "OTROS"; if (!req.file) { return res.status(400).json({ ok: false, message: "No se recibió archivo" }); } db.run( ` INSERT INTO uploads ( label, original_name, stored_name, mime_type, size ) VALUES (?, ?, ?, ?, ?) `, [ label, req.file.originalname, req.file.filename, req.file.mimetype, req.file.size ], function (err) { if (err) { return res.status(500).json({ ok: false, message: "Archivo subido pero no registrado en base de datos" }); } res.json({ ok: true, file: { id: this.lastID, label, original_name: req.file.originalname, stored_name: req.file.filename, mime_type: req.file.mimetype, size: req.file.size, url: `/uploads/${req.file.filename}` } }); } ); }); // Listar archivos app.get("/uploads-list", (req, res) => { db.all( `SELECT * FROM uploads ORDER BY id DESC`, [], (err, rows) => { if (err) { return res.status(500).json({ ok: false, message: "No se pudieron obtener los archivos" }); } const files = rows.map((row) => ({ ...row, url: `/uploads/${row.stored_name}` })); res.json({ ok: true, files }); } ); }); app.listen(PORT, () => { console.log(`Servidor G&T corriendo en http://localhost:${PORT}`); });
